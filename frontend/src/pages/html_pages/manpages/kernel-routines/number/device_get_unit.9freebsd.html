NAME
       device_get_unit — access the unit number of a device


SYNOPSIS
       #include <sys/param.h>
       #include <sys/bus.h>

       int
       device_get_unit(device_t dev);


DESCRIPTION
       Return the unit number of the device.

       device(9)
