NAME
       nvme_scan_ctrl_namespaces - Scan for namespaces in a controller


SYNOPSIS
       int nvme_scan_ctrl_namespaces (nvme_ctrl_t c , struct dirent ***ns );


ARGUMENTS
       c           Controller to scan

       ns          Pointer to array of dirents


RETURN
       number of entries in ns
