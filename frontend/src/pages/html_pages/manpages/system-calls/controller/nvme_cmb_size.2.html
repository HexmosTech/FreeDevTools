NAME
       nvme_cmb_size - Calculate size of the controller memory buffer


SYNOPSIS
       __u64 nvme_cmb_size (__u32 cmbsz );


ARGUMENTS
       cmbsz       Value from controller register NVME_REG_CMBSZ


RETURN
       size of controller memory buffer in bytes
