           NVME_AER_NOTICE_FW_ACT_STARTING ,
           NVME_AER_NOTICE_TELEMETRY ,

           NVME_AER_NOTICE_LBA_STATUS_ALERT ,
           NVME_AER_NOTICE_EG_EVENT ,
           NVME_AER_NOTICE_DISC_CHANGED

       };
