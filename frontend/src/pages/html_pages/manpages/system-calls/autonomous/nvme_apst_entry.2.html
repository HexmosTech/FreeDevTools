           NVME_APST_ENTRY_ITPS_SHIFT ,
           NVME_APST_ENTRY_ITPT_SHIFT ,
           NVME_APST_ENTRY_ITPS_MASK ,
           NVME_APST_ENTRY_ITPT_MASK

       };
