NAME
       struct nvme_firmware_slot - Firmware Slot Information Log


SYNOPSIS
       struct nvme_firmware_slot {
           __u8 afi;
           __u8 rsvd1[7];
           char frs[7][8];
           __u8 rsvd2[448];
        };


Members
       afi         Active Firmware Info

       rsvd1       Reserved

       frs         Firmware Revision for Slot

       rsvd2       Reserved
