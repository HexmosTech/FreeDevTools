NAME

           __u8 paid[16];
           __u8 rsvd16[14];
           __le16 attrl;
           __u8 vs[4064];
        };


Members
       paid        Performance attribute identifier

       rsvd16      Reserved

       attrl       Attribute Length

       vs          Vendor specific
