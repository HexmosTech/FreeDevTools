NAME
       struct nvme_zone_report - Report Zones Data Structure


SYNOPSIS
       struct nvme_zone_report {
           __le64 nr_zones;
           __u8 rsvd8[56];
           struct nvme_zns_desc    entries[];
        };


Members
       nr_zones    Number of descriptors in entries

       rsvd8       Reserved

       entries     Zoned namespace descriptors
