NAME
       CSAF::Util::Options - Options utility for CSAF


SYNOPSIS
           package My::CSAF::Options {

               use Moo;
               with 'CSAF::Util::Options';

               has enable_foo => (is => 'rw', required => 1);
               has make_bar => (is => 'rw');

           }

           package My::CSAF {

               use Moo;
               use My::CSAF::Options;

               has options => (is => 'lazy', build => 1);

               sub _build_options { My::CSAF::Options->new }

               sub my_job {

                   my $self = shift;

                   $self->options->configure(make_bar => $bar);

                   if ($self->options->enable_foo) {
                       [...]
                   }
               }

           }


DESCRIPTION
       CSAF::Util::Options is Moo role and utility for CSAF.

   ATTRIBUTES
       config_file
           Load config from YAML file.

   METHODS
       configure
           Configure an item.

               $options->configure(
                   foo => 'foo',
                   bar => 'bar',
                   baz => 'baz'
               );
