NAME
       socket_tcp6 - create a non-blocking IPv6 TCP/IP stream socket


SYNTAX
       #include <libowfat/socket.h>

       int socket_tcp6();


DESCRIPTION
       socket_tcp6  creates  a  non-blocking IPv6 TCP/IP stream socket and returns a file descriptor pointing to
       that socket.  If something goes wrong, socket_tcp6  returns  -1,  setting  errno  appropriately,  without
       allocating any resources.


EXAMPLE
         #include <libowfat/socket.h>

         int s;
         char ip[16];
         uint16 p;

         s = socket_tcp6();
         socket_bind6(s,ip,p);
         socket_connect6(s,ip,p,0);
