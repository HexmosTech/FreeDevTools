NAME
       MooseX::Runnable::Util::ArgParser - parse @ARGV for "mx-run"


VERSION
       version 0.10


SYNOPSIS
           my $parser = MooseX::Runnable::Util::ArgParser->new(
               argv => \@ARGV,
           );

           $parser->class_name;
           $parser->modules;
           $parser->include_paths;
           $parser->plugins;
           $parser->is_help;
           $parser->app_args;
