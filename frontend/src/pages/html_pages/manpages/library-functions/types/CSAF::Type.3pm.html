NAME
       CSAF::Type - Wrapper for all CSAF document types


SYNOPSIS
           use CSAF::Type;

           CSAF::Type->name(
               reference => {
                   url      => 'https://www.cve.org/CVERecord?id=CVE-2022-43634',
                   summary  => 'CVE-2022-43634',
                   category => 'external'
               }
           );

           CSAF::Type->new(
               name  => 'reference',
               value => {
                   url      => 'https://www.cve.org/CVERecord?id=CVE-2022-43634',
                   summary  => 'CVE-2022-43634',
                   category => 'external'
               }
           );


DESCRIPTION
   METHODS
       new ( name => $name, value => $value )
           Load and return the CSAF::Type::* class provided in name.

               CSAF::Type->new(
                   name  => 'reference',
                   value => {
                       url      => 'https://www.cve.org/CVERecord?id=CVE-2022-43634',
                       summary  => 'CVE-2022-43634',
                       category => 'external'
                   }
               );

       name ( $type => $value )
           Load and return the CSAF::Type::* class.

               CSAF::Type->name(
                   reference => {
                       url      => 'https://www.cve.org/CVERecord?id=CVE-2022-43634',
                       summary  => 'CVE-2022-43634',
                       category => 'external'
                   }
               );
