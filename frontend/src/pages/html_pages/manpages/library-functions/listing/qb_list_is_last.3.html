NAME
       qb_list_is_last - Tests whether list is the last entry in list head.


SYNOPSIS
       #include <qb/qblist.h>

       static int qb_list_is_last(
           const struct qb_list_head   *list,  /* the entry to test */
           const struct qb_list_head   *head   /* the head of the list */
       );


PARAMS
       list the entry to test

       head the head of the list


DESCRIPTION
STRUCTURES
       struct qb_list_head {
           struct   next;
           struct   prev;
       };


RETURN VALUE
       boolean true/false
