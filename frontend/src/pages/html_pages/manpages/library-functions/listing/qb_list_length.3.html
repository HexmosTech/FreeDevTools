NAME
       qb_list_length - Count the number of items in the list.


SYNOPSIS
       #include <qb/qblist.h>

       static int32_t qb_list_length(
           struct qb_list_head   *head   /* the head for your list. */
       );


PARAMS
       head the head for your list.


DESCRIPTION
STRUCTURES
       struct qb_list_head {
           struct   next;
           struct   prev;
       };


RETURN VALUE
       length of the list.
