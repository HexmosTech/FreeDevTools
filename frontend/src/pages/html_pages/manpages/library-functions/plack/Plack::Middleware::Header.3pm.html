NAME
       Plack::Middleware::Header - modify HTTP response headers


SYNOPSIS
         use Plack::Builder;

         my $app = sub {['200', [], ['hello']]};
         builder {
             enable 'Header',
               set => ['X-Plack-One' => '1'],
               append => ['X-Plack-Two' => '2'],
               unset => ['X-Plack-Three'];
             $app;
         };
