NAME
       SDL_VirtualJoystickTouchpadDesc - The structure that describes a virtual joystick touchpad.


SYNOPSIS
       #include <SDL3/SDL_joystick.h>

       typedef struct SDL_VirtualJoystickTouchpadDesc
       {
           Uint16 nfingers;    /**< the number of simultaneous fingers on this touchpad */
           Uint16 padding[3];
       } SDL_VirtualJoystickTouchpadDesc;


AVAILABILITY
       This struct is available since SDL 3.2.0.
