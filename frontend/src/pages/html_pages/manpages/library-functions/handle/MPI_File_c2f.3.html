NAME
       MPI_File_c2f -  Translates a C file handle to a Fortran file handle


SYNOPSIS
       #include "mpi.h"
       MPI_Fint MPI_File_c2f(MPI_File fh)


INPUT PARAMETERS
       fh     - C file handle (handle)


RETURN VALUE
       Fortran file handle (integer)
