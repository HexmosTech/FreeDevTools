NAME
       MPI_File_f2c -  Translates a Fortran file handle to a C file handle


SYNOPSIS
       #include "mpi.h"
       MPI_File MPI_File_f2c(MPI_Fint fh)


INPUT PARAMETERS
       fh     - Fortran file handle (integer)


RETURN VALUE
       C file handle (handle)
