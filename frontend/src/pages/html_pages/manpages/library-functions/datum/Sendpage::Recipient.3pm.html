NAME
       Recipient.pm - encapsulates the data of a single recipient


SYNOPSIS
           $recip=Sendpage::Recipient->new($config,$db,$name,$data);
           $alias=$recip->alias();
           @dests=@{ $recip->dests() };
           @data=@{ $recip->data() };
           $value=$recip->datum("field");
           $pin=$recip->pin();
           $pc=$recip->pc();
           $name=$recip->name();


DESCRIPTION
       This is a module for use in sendpage(1).
