NAME
       Bio::DB::HTS::Kseq::Record -- Entry from a Kseq iterator


SYNOPSIS
         while(my $r = $iter->next_seq()) {
           say $r->name;
           say $r->desc;
           say $r->seq;
           say $r->qual;
         }

   METHODS
       "name"  The name of a FASTA/Q record

       "desc"  The description from a FASTA/Q record

       "seq"   The sequence from a FASTA/Q record

       "qual"  The quality string from a FASTA/Q record
