NAME
       textout_justify_ex - Draws justified text within a region. Allegro game programming library.


SYNOPSIS
       #include <allegro.h>

       void  textout_justify_ex(BITMAP  *bmp, const FONT *f, const char *s, int x1, int x2, int y, int diff, int
       color, int bg);


DESCRIPTION
       Draws justified text within the region x1-x2. If the amount of spare  space  is  greater  than  the  diff
       value, it will give up and draw regular left justified text instead. Example:

          char *lines[] = {"Draws justified text",
                           "within the specified",
                           "x2-x1 area. But not",
                           "T H I S !", NULL};
          /* Show the justification marker. */
          vline(screen, 200, 0, SCREEN_H-1, makecol(0, 0, 0));
          /* Draw all the lines until we reach a NULL entry. */
          for (num = 0, y = 0; lines[num]; num++, y += text_height(font))
             textout_justify_ex(screen, font, lines[num], 0, 200,
                                y, 80, makecol(0, 0, 0),
                                makecol(255, 255, 255));
