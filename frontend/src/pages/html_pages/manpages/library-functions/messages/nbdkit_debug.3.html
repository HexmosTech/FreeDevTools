NAME
       nbdkit_debug, nbdkit_vdebug - print debugging messages for nbdkit


SYNOPSIS
        #include <nbdkit-plugin.h>

        void nbdkit_debug (const char *fs, ...);
        void nbdkit_vdebug (const char *fs, va_list args);


DESCRIPTION
       Use "nbdkit_debug" or "nbdkit_vdebug" to print debug messages from your plugin or filter.

       These functions send the debug message to stderr.  Debug messages are only emitted when the server is in
       verbose mode (nbdkit -v option)

       "nbdkit_debug" works like printf(3).  "nbdkit_vdebug" works like vprintf(3).

       For convenience both these functions preserve the value of "errno", and also support the glibc extension
       of a single %m in a format string expanding to strerror(errno), even on platforms that don't support that
       natively.


LANGUAGE BINDINGS
       In nbdkit-golang-plugin(3):

        import "libguestfs.org/nbdkit"
        nbdkit.Debug(msg)

       In nbdkit-ocaml-plugin(3):

        NBDKit.debug : ('a, unit, string, unit) format4 -> 'a

       In nbdkit-perl-plugin(3):

        Nbdkit::debug ($msg);

       In nbdkit-python-plugin(3):

        import nbdkit
        nbdkit.debug(str)

       In nbdkit-rust-plugin(3):

        use nbdkit::*;
        debug!(fs, ...);


HISTORY
       "nbdkit_debug" was present in nbdkit 0.1.0.
