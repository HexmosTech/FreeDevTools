NAME
       qb_loop_timer_del - Delete a timer that is still outstanding.


SYNOPSIS
       #include <qb/qbloop.h>

       int32_t qb_loop_timer_del(
           qb_loop_t            *l,   /* pointer to the loop instance */
           qb_loop_timer_handle  th   /* handle to delete the timer if needed. */
       );


PARAMS
       l  pointer to the loop instance

       th handle to delete the timer if needed.


DESCRIPTION
RETURN VALUE
       status (0 == ok, -errno == failure)
