NAME
       SDL_IMAGE_VERSION_ATLEAST - This macro will evaluate to true if compiled with SDL_image at least X.Y.Z.


HEADER FILE
       Defined in SDL3_image/SDL_image.h


SYNOPSIS
       #include "SDL3_image/SDL_image.h"

       #define SDL_IMAGE_VERSION_ATLEAST(X, Y, Z) .BI ((SDL_IMAGE_MAJOR_VERSION>=X)&&.BI     (SDL_IMAGE_MAJOR_VERSION > X || SDL_IMAGE_MINOR_VERSION >= Y) && .BI (SDL_IMAGE_MAJOR_VERSION>X||SDL_IMAGE_MINOR_VERSION>Y||SDL_IMAGE_MICRO_VERSION>=Z))
