NAME
       IM::Http - HTTP handler


SYNOPSIS
        use IM::Http;

        (rc, data) = http_process(spec, http_proxy, no_proxy)
            rc:
                0: success
                -1: failure


DESCRIPTION
       The IM::Http module handles HTTP.

       This modules is provided by IM (Internet Message).


EXAMPLES
        my($rc, $data) = http_process($spec, httpproxy(), noproxy())
        if ($rc < 0) {
            exit 1;
        }
        foreach (@$data) {
            print;
        }
