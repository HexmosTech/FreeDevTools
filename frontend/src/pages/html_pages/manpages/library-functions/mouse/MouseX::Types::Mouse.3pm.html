NAME
       MouseX::Types::Mouse - Types shipped with Mouse


SYNOPSIS
         package Foo;
         use Mouse;
         use MouseX::Types::Mouse qw( Int ArrayRef );

         has name => (
           is  => 'rw',
           isa => Str;
         );

         has ids => (
           is  => 'rw',
           isa => ArrayRef[Int],
         );

         1;
