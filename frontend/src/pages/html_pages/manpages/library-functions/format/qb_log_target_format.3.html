NAME
       qb_log_target_format - Format the callsite and timestamp info according to the format.


SYNOPSIS
       #include <qb/qblog.h>

       void qb_log_target_format(
           int32_t                   target,
           struct qb_log_callsite   *cs,
           struct timespec          *timestamp,
           const char               *formatted_message,
           char                     *output_buffer
       );


DESCRIPTION
       set using qb_log_format_set() It is intended to be used from your custom logger function.


STRUCTURES
       An instance of this structure is created for each log message with the message-id.

       struct qb_log_callsite {
           const char   *function;
           const char   *filename;
           const char   *format;
           uint8_t       priority;
           uint32_t      lineno;
           uint32_t      targets;
           uint32_t      tags;
           const char   *message_id;
       };
