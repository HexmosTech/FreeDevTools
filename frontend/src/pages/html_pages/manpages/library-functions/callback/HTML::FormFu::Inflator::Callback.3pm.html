NAME
       HTML::FormFu::Inflator::Callback - Callback inflator


VERSION
       version 2.07


SYNOPSIS
           $field->inflator('Callback')->callback( \&my_callback );

           ---
           elements:
             - type: Text
               name: foo
               inflators:
                 - type: Callback
                   callback: "main::my_inflator"

       Callback inflator.
