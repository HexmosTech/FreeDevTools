NAME
       Config::Model::TypeConstraints - Mouse type constraints for Config::Model


VERSION
       version 2.155


SYNOPSIS

        has 'some_dir' => (
           is => 'ro',
           isa => 'Config::Model::TypeContraints::Path',
           coerce => 1
        );
