NAME
       SDL_Sandbox - Application sandbox environment.


SYNOPSIS
       #include <SDL3/SDL_system.h>

       typedef enum SDL_Sandbox
       {
           SDL_SANDBOX_NONE = 0,
           SDL_SANDBOX_UNKNOWN_CONTAINER,
           SDL_SANDBOX_FLATPAK,
           SDL_SANDBOX_SNAP,
           SDL_SANDBOX_MACOS
       } SDL_Sandbox;


AVAILABILITY
       This enum is available since SDL 3.2.0.
