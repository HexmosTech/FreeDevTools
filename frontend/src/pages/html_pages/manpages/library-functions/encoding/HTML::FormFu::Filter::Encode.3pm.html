NAME
       HTML::FormFu::Filter::Encode - Encode/Decode Submitted Values


VERSION
       version 2.07


SYNOPSIS
          # in your config:
          elements:
             - type: Text
               filters:
                  - type: Encode
                    candidates:
                       - utf8
                       - Hebrew

          # if you want to encode the decoded string to something else
          elements:
             - type: Text
               filters:
                  - type: Encode
                    candidates:
                       - utf8
                       - Hebrew
                    encode_to: UTF-32BE
