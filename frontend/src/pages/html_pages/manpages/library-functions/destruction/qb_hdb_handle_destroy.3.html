NAME
       qb_hdb_handle_destroy - Request the destruction of the object.


SYNOPSIS
       #include <qb/qbhdb.h>

       int32_t qb_hdb_handle_destroy(
           struct qb_hdb   *hdb,        /* the database instance */
           qb_handle_t      handle_in   /* the handle */
       );


PARAMS
       hdb       the database instance

       handle_in the handle


DESCRIPTION
       When the refcount is 0, it will be destroyed.


STRUCTURES
       struct qb_hdb {
           uint32_t    handle_count;
           qb_array_t  handles;
           uint32_t    iterator;
           void      (*destructor)(void *);
           uint32_t    first_run;
       };


RETURN VALUE
       (0 == ok, -errno failure)
