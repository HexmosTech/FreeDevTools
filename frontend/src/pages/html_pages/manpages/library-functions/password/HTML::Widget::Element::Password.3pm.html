NAME
       HTML::Widget::Element::Password - Password Element


SYNOPSIS
           my $e = $widget->element( 'Password', 'foo' );
           $e->comment('(Required)');
           $e->fill(1);
           $e->label('Foo');
           $e->size(23);
           $e->value('bar');


DESCRIPTION
       Password Element.


METHODS
   retain_default
       If true, overrides the default behaviour, so that after a field is missing from the form submission, the
       xml output will contain the default value, rather than be empty.

   containerize
