NAME
       qb_ipcs_request_rate_limit - Limit the incoming request rate.


SYNOPSIS
       #include <qb/qbipcs.h>

       void qb_ipcs_request_rate_limit(
           qb_ipcs_service_t       *s,   /* service instance */
           enum qb_ipcs_rate_limit  rl   /* the new rate */
       );


PARAMS
       s  service instance

       rl the new rate


DESCRIPTION
STRUCTURES
       enum qb_ipcs_rate_limit {
             QB_IPCS_RATE_FAST;
             QB_IPCS_RATE_NORMAL;
             QB_IPCS_RATE_SLOW;
             QB_IPCS_RATE_OFF;
             QB_IPCS_RATE_OFF_2;
       };
