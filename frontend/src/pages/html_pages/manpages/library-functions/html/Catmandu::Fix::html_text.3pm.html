NAME
       Catmandu::Fix::html_text - keep only the textual data in the HTML


SYNOPSIS
          # keep only the text
          html_text()
          # returns:
          #  html: "<html>...</html>"

          # keep only the text but return an array_ref
          html_text(split:1)
          # returns:
          #  html:
          #    - <html>
          #    - ...
          #    - </html>
