NAME
       Catmandu::Fix::int - convert a value to an integer


SYNOPSIS
           # year => "2016"
           int(year)
           # year => 2016

           # foo => "bar-123baz"
           int(foo)
           # foo => -123

           # foo => ""
           int(foo)
           # foo => 0

           # foo => "abc"
           int(foo)
           # foo => 0

           # foo => []
           int(foo)
           # foo => 0

           # foo => ["a", "b", "c"]
           int(foo)
           # foo => 3

           # foo => {a => "b", c => "d"}
           int(foo)
           # foo => 2
