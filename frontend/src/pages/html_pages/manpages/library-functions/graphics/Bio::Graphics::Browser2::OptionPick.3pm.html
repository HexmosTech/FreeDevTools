NAME
       Bio::Graphics::Browser2::OptionPick -- Pick options


SYNOPSIS
        use Bio::Graphics::Browser2::OptionPick;
        my $picker = Bio::Graphics::Browser2::OptionPick->new($render_object);

        print start_html(),
         start_form(),
         $picker->('bgcolor','orange','blue'),
         end_form(),
         end_html();
