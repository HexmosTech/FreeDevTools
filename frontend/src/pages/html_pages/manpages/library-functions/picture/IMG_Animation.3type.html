NAME
       IMG_Animation - Animated image support


HEADER FILE
       Defined in SDL3_image/SDL_image.h


SYNOPSIS
       #include "SDL3_image/SDL_image.h"

       typedef struct IMG_Animation
       {
            int w, h;
            int count;
            SDL_Surface **frames;
            int *delays;
       } IMG_Animation;


DESCRIPTION
       Currently only animated GIFs and WEBP images are supported.
