NAME
       App::St - Simple Statistics


DESCRIPTION
       App::St provides the core functionality of the st-console application.


SYNOPSIS
         use App::St;

         my $st = App::St->new();

         while (<>) {
           chomp;
           next unless $st->validate($_);
           $st->process($_);
         }

         print $st->mean();
         print $st->stddev();
         print $st->sterr();


METHODS
   new(%options)
   validate($num)
   process($num)
   N
   sum
   mean
   stddev
   stderr
   percentile=<0..100>
   quartile=<0..4>
   min
   q1
   median
   q3
   max
