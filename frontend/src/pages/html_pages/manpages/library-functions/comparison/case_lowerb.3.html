NAME
       case_lowerb - compare strings case-insensitively


SYNTAX
       #include <libowfat/case.h>

       void case_lowerb(void* s,size_t len);


DESCRIPTION
       case_lowerb converts each 'A' to 'a', 'B' to 'b', ..., 'Z' to 'z' in s[0], s[1], ..., s[len].
