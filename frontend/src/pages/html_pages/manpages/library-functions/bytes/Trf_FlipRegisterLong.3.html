NAME
       Trf_FlipRegisterLong - Swap the bytes of all 4-byte words contained in the buffer.


SYNOPSIS
       #include <transform.h>

       void Trf_FlipRegisterLong
       (
            void *buffer,
            int length
       );


PARAMETERS
       void *buffer
              Data to swap.

       int length
              Length of buffer (in byte).


DESCRIPTION
       Swap the bytes of all 4-byte words contained in the buffer.
