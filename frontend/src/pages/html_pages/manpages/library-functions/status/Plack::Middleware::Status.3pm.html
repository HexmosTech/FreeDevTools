NAME
       Plack::Middleware::Status - Plack Middleware for mapping urls to status code-driven responses


VERSION
       version 1.101150


SYNOPSIS
           # app.psgi
           use Plack::Builder;
           my $app = sub {
               # ...
           };
           builder {
               enable 'Status', path => qr{/not-implemented}, status => 501;
               $app;
           };
