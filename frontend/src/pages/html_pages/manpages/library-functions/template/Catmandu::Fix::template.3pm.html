NAME
       Catmandu::Fix::template - add a value to the record based on a template


SYNOPSIS
          # Your record contains:
          #
          #  name: John
          #  age: 44

          template(message,"Mr [%name%] is [%age%] years old")

          # Result:
          #
          #  name: John
          #  age: 44
          #  message: Mr John is 44  years old
