NAME
       qb_hashtable_create - Create an unsorted map based on a hashtable.


SYNOPSIS
       #include <qb/qbmap.h>

       qb_map_t * qb_hashtable_create(
           size_t  max_size   /* maximum size of the hashtable*/
       );


PARAMS
       max_size maximum size of the hashtable


DESCRIPTION
RETURN VALUE
       the map instance
