NAME
       HTML::Display::Dump - dump raw HTML to the console


SYNOPSIS
         my $browser = HTML::Display->new(
           class => 'HTML::Display::Dump',
         );
         $browser->display("<html><body><h1>Hello world!</h1></body></html>");
