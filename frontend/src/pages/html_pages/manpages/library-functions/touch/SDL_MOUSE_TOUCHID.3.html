NAME
       SDL_MOUSE_TOUCHID - The SDL_TouchID for touch events simulated with mouse input.


SYNOPSIS
       #include <SDL3/SDL_touch.h>

       #define SDL_MOUSE_TOUCHID ((SDL_TouchID)-1)


AVAILABILITY
       This macro is available since SDL 3.2.0.
