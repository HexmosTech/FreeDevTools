NAME
       qb_list_replace - Replace old entry by new one.


SYNOPSIS
       #include <qb/qblist.h>

       static void qb_list_replace(
           struct qb_list_head   *old_one,  /* the element to be replaced */
           struct qb_list_head   *new_one   /* the new element to insert */
       );


PARAMS
       old_one the element to be replaced

       new_one the new element to insert


DESCRIPTION
STRUCTURES
       struct qb_list_head {
           struct   next;
           struct   prev;
       };
