NAME
       taia_uint - convert seconds into struct taia


SYNTAX
       #include <libowfat/taia.h>

       extern int taia_uint(struct taia* t,unsigned int secs);


DESCRIPTION
       taia_uint converts secs into a struct taia (setting the fractional part to zero).
