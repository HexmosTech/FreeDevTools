NAME
       waffle_x11_egl, waffle_x11_egl_display, waffle_x11_egl_config, waffle_x11_egl_context,
       waffle_x11_egl_window - Containers for underlying native X11/EGL objects


SYNOPSIS
       #include <waffle_x11_egl.h>

       struct waffle_x11_egl_display {
           Display *xlib_display;
           EGLDisplay egl_display;
       };

       struct waffle_x11_egl_config {
           struct waffle_x11_egl_display display;
           EGLConfig egl_config;
       };

       struct waffle_x11_egl_context {
           struct waffle_x11_egl_display display;
           EGLContext egl_context;
       };

       struct waffle_x11_egl_window {
           struct waffle_x11_egl_display display;
           XID xlib_window;
           EGLSurface egl_surface;
       };
