NAME
       XML::RPC::UA::Curl - XML::RPC useragent, using Curl


SYNOPSIS
           use XML::RPC::Fast;
           use XML::RPC::UA::Curl;

           my $rpc = XML::RPC::Fast->new(
               $uri,
               ua => XML::RPC::UA::Curl->new(
                   ua      => 'YourApp/0.1',
                   timeout => 3,
               ),
           );


DESCRIPTION
       Default syncronous useragent for XML::RPC::Fast


IMPLEMENTED METHODS
   new
   async = 0
   call
