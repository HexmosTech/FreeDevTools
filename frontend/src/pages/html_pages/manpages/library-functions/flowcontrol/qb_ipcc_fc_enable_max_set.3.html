NAME
       qb_ipcc_fc_enable_max_set - Set the maximum allowable flowcontrol value.


SYNOPSIS
       #include <qb/qbipcc.h>

       int32_t qb_ipcc_fc_enable_max_set(
           qb_ipcc_connection_t   *c,    /* connection instance */
           uint32_t                max   /* the max allowable flowcontrol value (1 or 2) */
       );


PARAMS
       c   connection instance

       max the max allowable flowcontrol value (1 or 2)


DESCRIPTION
NOTE
       the default is 1
