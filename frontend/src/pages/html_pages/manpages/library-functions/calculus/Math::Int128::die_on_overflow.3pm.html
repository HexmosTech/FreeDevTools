NAME
       Math::Int128::die_on_overflow - catch overflows when using Math::Int128


SYNOPSIS
         use Math::Int128 qw(uint128);
         use Math::Int128::die_on_overflow;

         my $number = uint128(2**64);
         say($number * $number); # overflow error!
