NAME
       Math::Int64::die_on_overflow - catch overflows when using Math::Int64


SYNOPSIS
         use Math::Int64 qw(uint64);
         use Math::Int64::die_on_overflow;

         my $number = uint64(2**32);
         say($number * $number); # overflow error!
