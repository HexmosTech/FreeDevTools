   Strftime Patterns
       %c (%A %d %b %Y %I:%M:%S %p) - date time format

          2008-02-05T18:30:30 = मंगळवार 05 फेब्रु 2008 06:30:30 PM
          1995-12-22T09:05:02 = शुक्रवार 22 डिसें 1995 09:05:02 AM
         -0010-09-15T04:44:23 = शनिवार 15 सप्टें -10 04:44:23 AM

       %x (%-d/%-m/%y) - date format

          2008-02-05T18:30:30 = %-d/%-m/08
          1995-12-22T09:05:02 = %-d/%-m/95
         -0010-09-15T04:44:23 = %-d/%-m/10

       %X (%I:%M:%S %p %Z) - time format

          2008-02-05T18:30:30 = 06:30:30 PM UTC
          1995-12-22T09:05:02 = 09:05:02 AM UTC
         -0010-09-15T04:44:23 = 04:44:23 AM UTC
