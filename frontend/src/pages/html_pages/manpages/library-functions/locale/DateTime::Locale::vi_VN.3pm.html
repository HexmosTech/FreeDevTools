   Strftime Patterns
       %c (%A, %d %B Năm %Y %T %Z) - date time format

          2008-02-05T18:30:30 = Thứ Ba, 05 tháng 2 Năm 2008 18:30:30 UTC
          1995-12-22T09:05:02 = Thứ Sáu, 22 tháng 12 Năm 1995 09:05:02 UTC
         -0010-09-15T04:44:23 = Thứ Bảy, 15 tháng 9 Năm -10 04:44:23 UTC

       %x (%d/%m/%Y) - date format

          2008-02-05T18:30:30 = 05/02/2008
          1995-12-22T09:05:02 = 22/12/1995
         -0010-09-15T04:44:23 = 15/09/-10

       %X (%T) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
