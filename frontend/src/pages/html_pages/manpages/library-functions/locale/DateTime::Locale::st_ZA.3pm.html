   Strftime Patterns
       %c (%a %-e %b %Y %T %Z) - date time format

          2008-02-05T18:30:30 = Bed %-e Kol 2008 18:30:30 UTC
          1995-12-22T09:05:02 = Hla %-e Tsh 1995 09:05:02 UTC
         -0010-09-15T04:44:23 = Moq %-e Leo -10 04:44:23 UTC

       %x (%d/%m/%Y) - date format

          2008-02-05T18:30:30 = 05/02/2008
          1995-12-22T09:05:02 = 22/12/1995
         -0010-09-15T04:44:23 = 15/09/-10

       %X (%T) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
