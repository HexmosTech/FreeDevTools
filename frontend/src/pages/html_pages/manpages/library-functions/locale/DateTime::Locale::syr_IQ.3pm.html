          2008-02-05T18:30:30 = ܬܠܬ ܫܒܛ  5 18:30:30 2008
          1995-12-22T09:05:02 = ܥܪܘ ܟܢܘܢ ܐ 22 09:05:02 1995
         -0010-09-15T04:44:23 = ܫܒܬܐ ܐܝܠܘܠ 15 04:44:23 -10

       %x (%m/%d/%y) - date format

          2008-02-05T18:30:30 = 02/05/08
          1995-12-22T09:05:02 = 12/22/95
         -0010-09-15T04:44:23 = 09/15/10

       %X (%H:%M:%S) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
