   Strftime Patterns
       %c (%d %b, %Y %I:%M:%S %p) - date time format

          2008-02-05T18:30:30 = 05 فبراير, 2008 06:30:30 م
          1995-12-22T09:05:02 = 22 ديسمبر, 1995 09:05:02 ص
         -0010-09-15T04:44:23 = 15 سبتمبر, -10 04:44:23 ص

       %x (%d %b, %Y) - date format

          2008-02-05T18:30:30 = 05 فبراير, 2008
          1995-12-22T09:05:02 = 22 ديسمبر, 1995
         -0010-09-15T04:44:23 = 15 سبتمبر, -10

       %X (%Z %I:%M:%S %p) - time format

          2008-02-05T18:30:30 = UTC 06:30:30 م
          1995-12-22T09:05:02 = UTC 09:05:02 ص
         -0010-09-15T04:44:23 = UTC 04:44:23 ص
