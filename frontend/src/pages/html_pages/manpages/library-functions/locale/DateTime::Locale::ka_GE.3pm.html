   Strftime Patterns
       %c (%Y წლის %d %B, %T) - date time format

          2008-02-05T18:30:30 = 2008 წლის 05 თებერვალი, 18:30:30
          1995-12-22T09:05:02 = 1995 წლის 22 დეკემბერი, 09:05:02
         -0010-09-15T04:44:23 = -10 წლის 15 სექტემბერი, 04:44:23

       %x (%m/%d/%Y) - date format

          2008-02-05T18:30:30 = 02/05/2008
          1995-12-22T09:05:02 = 12/22/1995
         -0010-09-15T04:44:23 = 09/15/-10

       %X (%T) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
