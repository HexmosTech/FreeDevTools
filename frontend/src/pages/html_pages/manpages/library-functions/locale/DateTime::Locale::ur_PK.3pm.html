   Strftime Patterns
       %c (و %H:%M:%S ت %d %B %Y) - date time format

          2008-02-05T18:30:30 = و 18:30:30 ت 05 فروری 2008
          1995-12-22T09:05:02 = و 09:05:02 ت 22 دسمبر 1995
         -0010-09-15T04:44:23 = و 04:44:23 ت 15 ستمبر -10

       %x (%d/%m/%Y) - date format

          2008-02-05T18:30:30 = 05/02/2008
          1995-12-22T09:05:02 = 22/12/1995
         -0010-09-15T04:44:23 = 15/09/-10

       %X (%H:%M:%S) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
