   Strftime Patterns
       %c (%a %e %b %Ey, %H:%M:%S) - date time format

          2008-02-05T18:30:30 = อ.  5 ก.พ. %Ey, 18:30:30
          1995-12-22T09:05:02 = ศ. 22 ธ.ค. %Ey, 09:05:02
         -0010-09-15T04:44:23 = ส. 15 ก.ย. %Ey, 04:44:23

       %x (%d/%m/%Ey) - date format

          2008-02-05T18:30:30 = 05/02/%Ey
          1995-12-22T09:05:02 = 22/12/%Ey
         -0010-09-15T04:44:23 = 15/09/%Ey

       %X (%H:%M:%S) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
