   Strftime Patterns
       %c (%a, %d-%b-%Y %X %z) - date time format

          2008-02-05T18:30:30 = вт, 05-лют.-2008 18:30:30 +0000
          1995-12-22T09:05:02 = пт, 22-груд.-1995 09:05:02 +0000
         -0010-09-15T04:44:23 = сб, 15-вер.--10 04:44:23 +0000

       %x (%d.%m.%y) - date format

          2008-02-05T18:30:30 = 05.02.08
          1995-12-22T09:05:02 = 22.12.95
         -0010-09-15T04:44:23 = 15.09.10

       %X (%T) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
