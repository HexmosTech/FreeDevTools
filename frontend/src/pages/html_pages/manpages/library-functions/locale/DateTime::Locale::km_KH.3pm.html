   Strftime Patterns
       %c (%A ថ្ងៃ %e ខែ %B ឆ្នាំ %Y, %H ម៉ោង %M នាទី %S វិនាទី) - date time format

          2008-02-05T18:30:30 = អង្គារ ថ្ងៃ  5 ខែ កុម្ភៈ ឆ្នាំ 2008, 18 ម៉ោង 30 នាទី 30 វិនាទី
          1995-12-22T09:05:02 = សុក្រ ថ្ងៃ 22 ខែ ធ្នូ ឆ្នាំ 1995, 09 ម៉ោង 05 នាទី 02 វិនាទី
         -0010-09-15T04:44:23 = សៅរ៍ ថ្ងៃ 15 ខែ កញ្ញា ឆ្នាំ -10, 04 ម៉ោង 44 នាទី 23 វិនាទី

       %x (%e %B %Y) - date format

          2008-02-05T18:30:30 =  5 កុម្ភៈ 2008
          1995-12-22T09:05:02 = 22 ធ្នូ 1995
         -0010-09-15T04:44:23 = 15 កញ្ញា -10

       %X (%H:%M:%S) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
