   Strftime Patterns
       %c (%A %d %B %Y %H:%M:%S) - date time format

          2008-02-05T18:30:30 = செவ்வாய் 05 பிப்ரவரி 2008 18:30:30
          1995-12-22T09:05:02 = வெள்ளி 22 டிசம்பர் 1995 09:05:02
         -0010-09-15T04:44:23 = சனி 15 செப்டம்பர் -10 04:44:23

       %x (%-d/%-m/%y) - date format

          2008-02-05T18:30:30 = %-d/%-m/08
          1995-12-22T09:05:02 = %-d/%-m/95
         -0010-09-15T04:44:23 = %-d/%-m/10

       %X (%H:%M:%S %Z) - time format

          2008-02-05T18:30:30 = 18:30:30 UTC
          1995-12-22T09:05:02 = 09:05:02 UTC
         -0010-09-15T04:44:23 = 04:44:23 UTC
