   Strftime Patterns
       %c (%A %d %b %Y %I:%M:%S %p) - date time format

          2008-02-05T18:30:30 = Tūsite 05 Fēp 2008 06:30:30 EA
          1995-12-22T09:05:02 = Falaite 22 Tīs 1995 09:05:02 HH
         -0010-09-15T04:44:23 = Tokonaki 15 Sēp -10 04:44:23 HH

       %x (%A %d %b %Y) - date format

          2008-02-05T18:30:30 = Tūsite 05 Fēp 2008
          1995-12-22T09:05:02 = Falaite 22 Tīs 1995
         -0010-09-15T04:44:23 = Tokonaki 15 Sēp -10

       %X (%I:%M:%S %p %Z) - time format

          2008-02-05T18:30:30 = 06:30:30 EA UTC
          1995-12-22T09:05:02 = 09:05:02 HH UTC
         -0010-09-15T04:44:23 = 04:44:23 HH UTC
