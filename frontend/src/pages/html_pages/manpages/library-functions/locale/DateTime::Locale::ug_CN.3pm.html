   Strftime Patterns
       %c (%a، %d-%m-%Y، %T) - date time format

          2008-02-05T18:30:30 = سە، 05-02-2008، 18:30:30
          1995-12-22T09:05:02 = جۈ، 22-12-1995، 09:05:02
         -0010-09-15T04:44:23 = شە، 15-09--10، 04:44:23

       %x (%a، %d-%m-%Y) - date format

          2008-02-05T18:30:30 = سە، 05-02-2008
          1995-12-22T09:05:02 = جۈ، 22-12-1995
         -0010-09-15T04:44:23 = شە، 15-09--10

       %X (%T) - time format

          2008-02-05T18:30:30 = 18:30:30
          1995-12-22T09:05:02 = 09:05:02
         -0010-09-15T04:44:23 = 04:44:23
