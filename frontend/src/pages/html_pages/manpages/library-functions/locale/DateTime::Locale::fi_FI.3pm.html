   Strftime Patterns
       %c (%a %e. %Bta %Y %H.%M.%S) - date time format

          2008-02-05T18:30:30 = ti  5. helmikuutata 2008 18.30.30
          1995-12-22T09:05:02 = pe 22. joulukuutata 1995 09.05.02
         -0010-09-15T04:44:23 = la 15. syyskuutata -10 04.44.23

       %x (%d.%m.%Y) - date format

          2008-02-05T18:30:30 = 05.02.2008
          1995-12-22T09:05:02 = 22.12.1995
         -0010-09-15T04:44:23 = 15.09.-10

       %X (%H.%M.%S) - time format

          2008-02-05T18:30:30 = 18.30.30
          1995-12-22T09:05:02 = 09.05.02
         -0010-09-15T04:44:23 = 04.44.23
