       [SimkaMerge options]

       -nb-cores
              (1 arg) :    number of cores  [default '0']

       -verbose
              (1 arg) :    verbosity level  [default '1']

       -version
              (0 arg) :    version

       -help  (0 arg) :    help

       -nb-cores
              (1 arg) :    nb cores

       -kmer-size
              (1 arg) :    kmer size

       -in    (1 arg) :    input filename

       -out-tmp-simka
              (1 arg) :    tmp output

       -partition-id
              (1 arg) :    bank name

       -nb-cores
              (1 arg) :    bank name

       -max-memory
              (1 arg) :    bank name

       -kmer-shannon-index (1 arg) :
              bank name

       -simple-dist
              (0 arg) :    compute simple distances

       -complex-dist
              (0 arg) :    compute complex distances
