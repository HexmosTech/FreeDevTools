NAME
       pypi-download - downloads tarballs from PyPI


SYNOPSIS
       pypi-download PACKAGE_NAME [options]


OPTIONS
       -h, --help
              show this help message and exit

       --verbose=VERBOSE
              verbosity level

       --release=RELEASE
              specify a particular release

       --allow-unsafe-download
              allow unsafe downloads
