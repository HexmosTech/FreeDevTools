NAME
       ncclimo - netCDF Climatology Generator


SYNTAX
       ncclimo  [-3]  [-4]  [-5] [-6] [-7] [-a wnt_md] [-C clm_md] [-c caseid] [--cmp_sng= cmp_sng] [-d dbg_lvl]
       [--d2f] [--dpt_fl dpt_fl] [-E yr_prv] [-e  yr_end]  [--mth_end=  mth_end]  [-f  fml_nm]  [--fl_fmt=  fmt]
       [--glb_avg]  [--glb_stt]  avg|sum]  [-h  hst_nm]  [-i  drc_in] [-j job_nbr] [-L dfl_lvl] [-l lnk_flg] [-m
       mdl_nm] [-n nco_usr] [--no_cll_msr] [--no_frm_trm] [--no_ntv_tms] [--no_stg_grd] [--no_stdin] [--npo] [-O
       drc_rgr] [-o drc_out] [-p par_typ] [--qnt= qnt_prc] [-R rgr_opt]  [-r  rgr_map]  [--rm_ntv=  rm_ntv]  [-S
       yr_prv]  [-s yr_srt] [--mth_srt= mth_srt] [--seasons= csn_lst] [--sgs_frc= sgs_frc] [--split] [--sum_scl=
       sum_scl] [-t thr_nbr] [--tpd_out= tpd_out] [--uio] [-v var_lst] [--var_xtr  var_xtr]  [--version]  [--vrt
       vrt_crd]  [--vrt_xtr= vrt_xtr] [-X drc_xtn] [-x drc_prv] [--xcl_var] [-Y rgr_xtn] [-y rgr_prv] [--ypf_max
       ypf_max]


DESCRIPTION
       ncclimo produces climatological monthly means, seasonal means, annual mean, and  optionally  regrids  all
       these files.
