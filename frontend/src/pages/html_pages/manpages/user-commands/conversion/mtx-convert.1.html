NAME
        mtx-convert - ConTeXT Graphic Conversion Helpers


SYNOPSIS
       mtxrun --script convert [ OPTIONS ...  ] [ FILENAMES ]


DESCRIPTION
       ConTeXT Graphic Conversion Helpers


OPTIONS
       --convertall
              convert all graphics on path

       --inputpath=string
              original graphics path

       --outputpath=string
              converted graphics path

       --watch
              watch folders

       --force
              force conversion (even if older)

       --delay
              time between sweeps
