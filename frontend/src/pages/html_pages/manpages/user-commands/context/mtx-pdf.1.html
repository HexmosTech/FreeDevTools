NAME
        mtx-pdf - ConTeXt PDF Helpers


SYNOPSIS
       mtxrun --script pdf [ OPTIONS ...  ] [ FILENAMES ]


DESCRIPTION
       ConTeXt PDF Helpers


OPTIONS
       --info show some info about the given file

       --metadata
              show metadata xml blob

       --formdata
              show formdata

       --pretty
              replace newlines in metadata

       --fonts
              show used fonts (--detail)

       --object
              show object

       --links
              show links

       --sign sign document (assumes signature template)

       --verify
              verify document

       --detail
              print detail to the console
