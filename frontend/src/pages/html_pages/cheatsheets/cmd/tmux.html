Tmux

[main]
name=tmux
desc=tmux shortcut
[cheats]
bash - start new with session name      = tmux new -s <SESSION_NAME>
bash - attach to a recent session       = tmux a # (or at, or attach)
bash - attach to a named session        = tmux a -t <SESSION_NAME>
bash - list sessions                    = tmux ls
bash - kill session                     = tmux kill-session -t <SESSION_NAME>
bash - kill all the tmux sessions       = tmux ls | grep : | cut -d. -f1 | awk '{print substr($1, 0, length($1)-1)}' | xargs kill
C-b - start new session                 = C-b :new<CR>
C-b - list sessions                     = C-b s
C-b - rename session                    = C-b $
C-b - detach session                    = C-b d
C-b - create new window                 = C-b c
C-b - list windows                      = C-b w
C-b - next window                       = C-b n
C-b - previous window                   = C-b p
C-b - find window                       = C-b f
C-b - name window                       = C-b ,
C-b - kill window                       = C-b &
C-b - switch window with number         = C-b 1 ...
C-b - vertical split                    = C-b %%
C-b - horizontal split                  = C-b "
C-b - swap panes                        = C-b o
C-b - show pane number                  = C-b q
C-b - kill pane                         = C-b x
C-b - break pane into window (e.g. to select text by mouse to copy) = C-b +
C-b - restore pane from window          = C-b -
C-b - resize pane down by x cells       = C-b :resize-pane -D <X> 
C-b - resize pane up by x cells         = C-b :resize-pane -U <X> 
C-b - resize pane left by x cells       = C-b :resize-pane -L <X> 
C-b - resize pane right by x cells      = C-b :resize-pane -R <X> 
C-b - resize pane <Y> right by x cells  = C-b :resize-pane -t <PANE_ID> -R <X> 
C-b - change layout                     = C-b space
C-b - display big block                 = C-b t
C-b - list shortcuts                    = C-b ?
C-b - show prompt                       = C-b :
